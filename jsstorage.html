// ==================== STORAGE KEYS ==================== const STORAGE_KEYS = { PRODUCTS: 'gm_products', CLIENT_PRODUCTS: 'gm_client_products', CATEGORIES: 'gm_categories', ORDERS: 'gm_orders', CART: 'gm_cart', LAST_UPDATE: 'gm_last_update' }; // ==================== DATA STRUCTURE ==================== let sharedData = { products: [], clientProducts: [], categories: {}, orders: [], cart: [] }; // ==================== INITIALIZE STORAGE ==================== function initStorage() { // Load all data from localStorage sharedData.products = JSON.parse(localStorage.getItem(STORAGE_KEYS.PRODUCTS)) || []; sharedData.clientProducts = JSON.parse(localStorage.getItem(STORAGE_KEYS.CLIENT_PRODUCTS)) || []; sharedData.categories = JSON.parse(localStorage.getItem(STORAGE_KEYS.CATEGORIES)) || { 'Electronics': 'https://cdn-icons-png.flaticon.com/512/3659/3659898.png', 'Fashion': 'https://cdn-icons-png.flaticon.com/512/6174/6174039.png', 'Home': 'https://cdn-icons-png.flaticon.com/512/2543/2543366.png', 'Sports': 'https://cdn-icons-png.flaticon.com/512/2965/2965275.png', 'Client Products': 'https://cdn-icons-png.flaticon.com/512/3135/3135715.png' }; sharedData.orders = JSON.parse(localStorage.getItem(STORAGE_KEYS.ORDERS)) || []; sharedData.cart = JSON.parse(localStorage.getItem(STORAGE_KEYS.CART)) || []; // Set last update timestamp updateTimestamp(); } // ==================== SAVE ALL DATA ==================== function saveAllData() { localStorage.setItem(STORAGE_KEYS.PRODUCTS, JSON.stringify(sharedData.products)); localStorage.setItem(STORAGE_KEYS.CLIENT_PRODUCTS, JSON.stringify(sharedData.clientProducts)); localStorage.setItem(STORAGE_KEYS.CATEGORIES, JSON.stringify(sharedData.categories)); localStorage.setItem(STORAGE_KEYS.ORDERS, JSON.stringify(sharedData.orders)); localStorage.setItem(STORAGE_KEYS.CART, JSON.stringify(sharedData.cart)); updateTimestamp(); // Trigger storage event for cross-tab sync triggerStorageSync(); } // ==================== UPDATE TIMESTAMP ==================== function updateTimestamp() { localStorage.setItem(STORAGE_KEYS.LAST_UPDATE, Date.now().toString()); } // ==================== TRIGGER SYNC ==================== function triggerStorageSync() { // This will trigger storage event in other tabs localStorage.setItem('gm_sync_trigger', Date.now().toString()); } // ==================== GET DATA ==================== function getData(type) { return sharedData[type] || []; } // ==================== UPDATE DATA ==================== function updateData(type, newData) { sharedData[type] = newData; saveAllData(); } // ==================== ADD ITEM ==================== function addItem(type, item) { sharedData[type].push(item); saveAllData(); return item; } // ==================== UPDATE ITEM ==================== function updateItem(type, index, newItem) { if (index >= 0 && index < sharedData[type].length) { sharedData[type][index] = newItem; saveAllData(); return true; } return false; } // ==================== DELETE ITEM ==================== function deleteItem(type, index) { if (index >= 0 && index < sharedData[type].length) { sharedData[type].splice(index, 1); saveAllData(); return true; } return false; } // ==================== SYNC LISTENER ==================== function addSyncListener(callback) { window.addEventListener('storage', function(event) { if (event.key === STORAGE_KEYS.LAST_UPDATE || event.key === 'gm_sync_trigger') { // Reload all data initStorage(); if (callback) callback(sharedData); } }); } // Initialize on load initStorage();
